30 void generatePath(const std::vector<float>& renderData,
31 juce::Rectangle<float> fftBounds,
32 int fftSize,
33 float binWidth,
34 float negativeInfinity)
35 {
36 auto top = fftBounds.getY();
37 auto bottom = fftBounds.getBottom();
38 // auto bottom = fftBounds.getHeight();
39 auto width = fftBounds.getWidth();
40
41 int numBins = (int)fftSize / 2;
42
43 PathType p;
44 p.preallocateSpace(3 * (int)fftBounds.getWidth());
45
46 auto map = [bottom, top, negativeInfinity](float v)
47 {
48 return juce::jmap(v,
49 negativeInfinity, 0.f,
50 // float(bottom+10),
51 bottom,
52 top);
53
54 };
55
56 auto y = map(renderData[0]);
57
58 // jassert( !std::isnan(y) && !std::isinf(y) );
59 if( std::isnan(y) || std::isinf(y) )
60 y = bottom;
61
62 p.startNewSubPath(0, y);
63
64 const int pathResolution = 2; //you can draw line-to's every 'pathResolution' pixels.
65
66 for( int binNum = 1; binNum < numBins; binNum += pathResolution )
67 {
68 y = map(renderData[binNum]);
69
70 // jassert( !std::isnan(y) && !std::isinf(y) );
71
72 if( !std::isnan(y) && !std::isinf(y) )
73 {
74 auto binFreq = binNum * binWidth;
75 auto normalizedBinX = juce::mapFromLog10(binFreq, 20.f, 20000.f);
76 int binX = std::floor(normalizedBinX * width);
77 p.lineTo(binX, y);
78 }
79 }
80
81 pathFifo.push(p);
82 }
83
84 int getNumPathsAvailable() const

87 }
88
89 bool getPath(PathType& path)

92 }
93private:
