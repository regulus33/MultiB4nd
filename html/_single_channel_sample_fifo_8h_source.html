15enum Channel
16{
17 Right, //effectively 0
18 Left //effectively 1
19};
20
21template<typename BlockType>
22struct SingleChannelSampleFifo
23{
24 SingleChannelSampleFifo(Channel ch) : channelToUse(ch)
25 {
26 prepared.set(false);
27 }
28
29 void update(const BlockType& buffer)
30 {
31 jassert(prepared.get());
32 jassert(buffer.getNumChannels() > channelToUse );
33 auto* channelPtr = buffer.getReadPointer(channelToUse);
34
35 for( int i = 0; i < buffer.getNumSamples(); ++i )
36 {
37 pushNextSampleIntoFifo(channelPtr[i]);
38 }
39 }
40
41 void prepare(int bufferSize)
42 {
43 prepared.set(false);
44 size.set(bufferSize);
45
46 bufferToFill.setSize(1, //channel
47 bufferSize, //num samples
48 false, //keepExistingContent
49 true, //clear extra space
50 true); //avoid reallocating
51 audioBufferFifo.prepare(1, bufferSize);
52 fifoIndex = 0;
53 prepared.set(true);
54 }
55 //==============================================================================
56 int getNumCompleteBuffersAvailable() const { return audioBufferFifo.getNumAvailableForReading(); }
57 bool isPrepared() const { return prepared.get(); }
58 int getSize() const { return size.get(); }
59 //==============================================================================
60 bool getAudioBuffer(BlockType& buf) { return audioBufferFifo.pull(buf); }
61private:
62 Channel channelToUse;
63 int fifoIndex = 0;
64 Fifo<BlockType> audioBufferFifo;
65 BlockType bufferToFill;
66 juce::Atomic<bool> prepared = false;
67 juce::Atomic<int> size = 0;
68
69 void pushNextSampleIntoFifo(float sample)
70 {
71 if (fifoIndex == bufferToFill.getNumSamples())
72 {
73 auto ok = audioBufferFifo.push(bufferToFill);
74
75 juce::ignoreUnused(ok);
76
77 fifoIndex = 0;
78 }
79
80 bufferToFill.setSample(0, fifoIndex, sample);
81 ++fifoIndex;
82 }
83};
